#########################################################################

OpenSCAD_DocTest API

-------------------------------------------------------------------------
This is the api of OpenSCAD_DocTest lib. It is created using function

doctest_tool_test(mode=13) in file openscad_doctest_help.scad. The output

to the OpenSCAD console is copied and pasted into LibraOffice Writer, and

saved to this html file after minor editing. This is a good demo of using

OpenSCAD_DocTest for documentation.


######################################################################### 
doctest()

-----------------------------------------------------------------
doctest ( doc,recs=[],ops=[],scope=[] )

| Doc and test tool for OpenSCAD functions. 
| 
| doc: array of 5 strings:
|
|   [ func_name, arg_name(s), return_type, tags, doc_content ]
|
|   e.g, ["addz","p,z","array","Array","doc..."]
|
| recs: array of test cases, each is:
|
|   [ arg_string, result, wanted, test_option ]
|
|   e.g, [ "[3,4],2", addz([3,4],2), [3,4,2] ]
|
|   where test_option is an optional array:
|
|     [ "rem", ..., "asstr", true|false ]  
| 
| ops: options for all test cases in this call:
|
|   [ "mode", 13      // Chk doctest_help()
|   , "asstr", false  // test all as string
|   , "prefix", "|" // prefix for each line
|   ]
| 
| scope: scope used for test cases. Useful especially when
|    the var value is long and need to be used multiple times.
|    It is in a hash format, e.g., ["point",[-1,2,3]]
|
| Tags: Doc,Test

dt＿fmt ( x,[s,n,a,ud,fixedspace] )=string

| Given an variable (x), and formattings for string (s), number (n),
| array (a) or undef (ud), return a formatted string of x. If
| fixedspace is set to true, replace spaces with html space. The
| formatting is entered in "(prefix),(suffix)" pattern. For
| example, if x is a number, setting n="{,}" converts x to "{x}".
|
| Tags: String
|
|Usage:
|
|dt_fmt(3.45): 3.45
|dt_fmt([''a'',[3,''b''], 4]): ["a", [3, "b"], 4]
|dt_fmt(''a string with     5 spaces''): "astringwith5spaces"
|dt_fmt(false): false

dt＿hash ( h,k, notfound=undef, if_v=undef, then_v=undef, else_v=undef )=Any ( tested:24 )

| Given an array (h, arranged like h=[k,v,k2,v2 ...] to serve
| as a hash) and a key (k), return a v from h: dt_hash(h,k)=v.
| If key not found, or value not found (means k is the last item
| in h), return notfound. When if_v is defined, check if v=if_v.
| If yes, return then_v. If not, return else_v if defined, or v
| if not.
|
| Tags: dt_hash
|
|Tests:
|
|> h= [-50, 20, "xy", [0, 1], 2.3, 5, [0, 1], 10, true, 1, 1, false, -5]; 
|> dt_hash( h, -50 )= 20
|> dt_hash( h, "xy" )= [0, 1]
|> dt_hash( h, 2.3 )= 5
|> dt_hash( h, [0,1] )= 10
|> dt_hash( h, true )= 1
|> dt_hash( h, 1 )= false
|> dt_hash( h, "xx" )= undef
|## key found but it's in the last item without value:
|> dt_hash( h, -5 )= undef
|## notfound is given: 
|> dt_hash( h, -5,"df" )= "df" // value not found
|> dt_hash( h, -50, "df" )= 20
|> dt_hash( h, -100, "df" )= "df"
|## Other cases:
|> dt_hash( [], -100 )= undef
|> dt_hash( true, -100 )= undef
|> dt_hash( [[undef,10]], undef )= undef
|  
|## Note below: you can ask for a default upon a non-hash !!
|> dt_hash( true, -100, "df" )= "df"
|> dt_hash( true, -100  )= undef
|> dt_hash( 3, -100, "df" )= "df"
|> dt_hash( "test", -100, "df" )= "df"
|> dt_hash( false, 5, "df" )= "df"
|> dt_hash( 0, 5, "df" )= "df"
|> dt_hash( undef, 5, "df" )= "df"
|  
|## New 20140528: if_v, then_v, else_v 
|> dt_hash( h, 2.3, if_v=5, then_v="got_5" )= "got_5"
|> dt_hash( h, 2.3, if_v=6, then_v="got_5" )= 5
|> dt_hash( h, 2.3, if_v=6, then_v="got_5", else_v="not_5" )= "no_5"

dt＿isequal ( a,b )=T|F

| Given 2 numbers a and b, return true if they are determined to
| be the same value. This is needed because sometimes a calculation
| that should return 0 ends up with something like 0.0000000000102.
| So it fails to compare with 0.0000000000104 (another shoulda-been
| 0) correctly. This function checks if abs(a-b) is smaller than the
| predifined DT_ZERO constant (1e-13) and return true if it is.
|
| Tags: Inspect
